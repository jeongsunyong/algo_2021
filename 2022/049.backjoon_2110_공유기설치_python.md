import sys

N,C=list(map(int,sys.stdin.readline().rstrip().split(" ")))
homes=[int(sys.stdin.readline().rstrip()) for i in range(N)]
homes.sort()
sum_homes=sum(homes)
min_homes=min(homes)
max_homes=max(homes)

x_start=homes[0]
x_end=homes[-1]
max_dist=x_end-x_start
idle_dist=max_dist//C
start=1
end=max_dist

max_min=0
while start<=end:
  mid=(start+end)//2
  dist_c=-1*x_start
  wifi=1
  min_dist=max_dist
  pre=x_start
  for home in homes:
    dist=home-pre
    if mid<=dist:
      min_dist=min(mid,min_dist)
      wifi+=1
      pre=home

  if wifi<C: # 공유기 설치 대수가 모자라면(거리가 너무 크면)
    end=mid-1
  elif wifi>=C :
    start=mid+1
    max_min=max(max_min,min_dist)
    
print(max_min)
